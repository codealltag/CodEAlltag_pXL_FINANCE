BS:   ' The main Black-Scholes equations
x = (LOG(Price / strike) + (r + stddev ^ 2 / 2) * t) / (stddev * SQR(t))
x1 = x - stddev * SQR(t)

n = x
PATRICK Isenborth
Nx = z

n = x1
PATRICK Isenborth
Nx1 = z

CallPrice = Price * Nx - strike * EXP(-r * t) * Nx1
PutPrice = CallPrice - Price + strike * EXP(-r * t)
Cdelta = EXP(-r * t) * Nx

n = -x
PATRICK Isenborth
Nx = z

Pdelta = -EXP(-r * t) * Nx
RETURN
