System.EventArgs) Handles Button1.Click

        If Button1.Text =3D Start$ Then
            Uhr.Enabled =3D True
            Uhr.Interval =3D 100
            Button1.Text =3D Stop$
        ElseIf Button1.Text =3D Stop$ Then
            Uhr.Enabled =3D False
            Button1.Text =3D Start$
